150 void MLX90621::precalculate_constants() {
156  // Parameters for ambient temperature calculations
157  // See 7.3.1 - Calculation of absolute chip temperature Ta (sensor temperature)
158  resolution_comp = pow(2.0, (3 - get_resolution()));
159  k_t1_scale = (int16_t) (eeprom_buffer[KT_SCALE] & 0xF0) >> 4;
160  k_t2_scale = (int16_t) (eeprom_buffer[KT_SCALE] & 0x0F) + 10;
161 
162  v_th = (float) twos_16(eeprom_buffer[VTH_H], eeprom_buffer[VTH_L]);
163  v_th = v_th / resolution_comp;
164 
165  k_t1 = (float) twos_16(eeprom_buffer[KT1_H], eeprom_buffer[KT1_L]);
166  k_t1 /= (pow(2, k_t1_scale) * resolution_comp);
167 
168  k_t2 = (float) twos_16(eeprom_buffer[KT2_H], eeprom_buffer[KT2_L]);
169  k_t2 /= (pow(2, k_t2_scale) * resolution_comp);
170 
171  // Parameters for object temperature calculations
172  // See 7.3.3 - Calculation of To
173  tgc = (float) twos_8(eeprom_buffer[CAL_TGC]) / 32.0;
174  emissivity = unsigned_16(eeprom_buffer[CAL_EMIS_H], eeprom_buffer[CAL_EMIS_L]) / 32768.0;
175  a_common = twos_16(eeprom_buffer[CAL_ACOMMON_H], eeprom_buffer[CAL_ACOMMON_L]);
176  a_i_scale = (int16_t)(eeprom_buffer[CAL_AI_SCALE] & 0xF0) >> 4;
177  b_i_scale = (int16_t) eeprom_buffer[CAL_BI_SCALE] & 0x0F;
178 
179  alpha_cp = unsigned_16(eeprom_buffer[CAL_alphaCP_H], eeprom_buffer[CAL_alphaCP_L])/ (pow(2.0, eeprom_buffer[CAL_A0_SCALE]) * resolution_comp);
180  a_cp = (float) twos_16(eeprom_buffer[CAL_ACP_H], eeprom_buffer[CAL_ACP_L]) / resolution_comp;
181  b_cp = (float) twos_8(eeprom_buffer[CAL_BCP]) / (pow(2.0, (float)b_i_scale) * resolution_comp);
182 }

281 float MLX90621::get_ambient_temperature(){
288  check_configuration();
289  float ptat = float(get_PTAT());
290 
291  // MLX90621 Datasheet - 7.3.1 - Calculation of absolute chip temperature Ta
292  float ambient_temperature = ((-k_t1 + sqrt((k_t1 * k_t1) - (4 * k_t2 * (v_th - ptat)))) / (2 * k_t2)) + 25.0;
293  return ambient_temperature;

418 float MLX90621::calculate_pixel(uint8_t pixel_num, int ir_data[]){
427  // Grab the calibration and offset values for the individual pixel
428  double a_ij = ((float) a_common + eeprom_buffer[pixel_num] * pow(2.0, a_i_scale)) / resolution_comp;
429  double b_ij = (float) twos_8(eeprom_buffer[0x40 + pixel_num]) / (pow(2.0, b_i_scale) * resolution_comp);
430  double v_ir_off_comp = (float) ir_data[pixel_num] - (a_ij + b_ij * (ambient - 25.0));
431  double v_ir_tgc_comp = (float) v_ir_off_comp - tgc * v_cp_off_comp;
432 
433  double alpha_ij = ((float) unsigned_16(eeprom_buffer[CAL_A0_H], eeprom_buffer[CAL_A0_L]) / pow(2.0, (float) eeprom_buffer[CAL_A0_SCALE]));
434  alpha_ij += ((float) eeprom_buffer[0x80 + pixel_num] / pow(2.0, (float) eeprom_buffer[CAL_DELTA_A_SCALE]));
435  alpha_ij = alpha_ij / resolution_comp;
436 
437  //ksta = (float) twos_16(eeprom_buffer[CAL_KSTA_H], eeprom_buffer[CAL_KSTA_L]) / pow(2.0, 20.0);
438  //alpha_comp = (1 + ksta * (Tambient - 25.0)) * (alpha_ij - tgc * alpha_cp);
439  double alpha_comp = (alpha_ij - tgc * alpha_cp); // For my MLX90621 the ksta calibrations were 0
440  // so I can ignore them and save a few cycles
441  float v_ir_comp = v_ir_tgc_comp / emissivity;
442  double temperature = pow((v_ir_comp / alpha_comp) + tak4, 1.0 / 4.0) - 273.15;
443 
444  return temperature;
